#!/bin/bash

clear

read -p "Informe o valor 1: " n1
read -p "Informe o valor 2: " n2

echo "1 - Ler valores"
echo "2 - Somar"
echo "3 - Subtrair"
echo "4 - Multiplicar"
echo "5 - Dividir"
echo "Informe a opção desejada"
read op

case $op in
	1)
		echo "Valor 1: $n1"
		echo "Valor 2: $n2"
		;;	
	2)
		soma=$(expr $n1 + $n2)
		echo "Soma = $soma"
		;;
	3)
		sub=$(expr $n1 - $n2)
		echo "Subtração = $sub"
		;;
	4)
		mul=$(expr $n1 \* $n2)
		echo "Multiplicação = $mul"
		;;
	5)
		div=$(echo "scale=2; $n1 / $n2" | bc)
		echo "Divisão = $div"
		;;
	*)
		echo "Opção inválida"
esac
